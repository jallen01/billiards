\subsection{Algorithm}

\frame{
  \frametitle{Algorithm}

  \begin{align*}
    dx_n& = \bigcap_{i=0}^n \left( \frac{i}{1 + \sum_{j=0}^i n_j}, \frac{1}{-1 + \sum_{j=0}^i n_j} \right)\\
    \delta_n& = \bigcap_{i=0}^n \left( i-dx_{n, max} (1 + \sum_{j=0}^i n_j), i-dx_{n, min} (1 + \sum_{j=0}^i n_j) \right)
  \end{align*}
}
