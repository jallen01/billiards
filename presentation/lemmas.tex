\subsection{Tiling}

\frame{
  \frametitle{Representing Collision Strings}

  \begin{itemize}
    \item Reflect squares about each side to create a tiling
    \item Solutions become lines in the plane
    \item Intersections become places where collisions occur
  \end{itemize}
}
