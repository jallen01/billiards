\frame{
	\frametitle{Second Indexing Theorem}

	\begin{description}
		\item[$\beta_{max}$:] maximum $\beta$ value in $\alpha$: $\max_i \beta_i$\\
		\item[$\beta_{min}$:] minimum $\beta$ value in $\alpha$: $\min_i \beta_i$
	\end{description}

	\begin{theorem}
		Every valid collision sequence has the following property

		\[ \beta_{max} - \beta_{min} \le 1 \]
	\end{theorem}

}

\frame{
	\frametitle{Second Indexing Theorem}

	\begin{example}
		\begin{itemize}
			\item $\alpha = hvvhhvvvhvvhhvhhvvvvhvvvh \qquad \beta = (2, 0, 3, 2, 0, 1, 2, 4, 3)$
			\item $\alpha = hvvvvhvvvvhvhvvvhvvvvvvvh \qquad \beta = (4, 4, 1, 3, 7)$
			\item $\alpha = hvvvvhvvvvhvvvhvvvvhvvvvh \qquad \beta = (4, 4, 3, 4, 4)$
		\end{itemize}
	\end{example}
}
